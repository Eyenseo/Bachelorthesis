\newcommand\minco[5][cpp]{
  {
    \ifthenelse{\equal{#2}{}}
    {}
    {\noindent\texttt{#2}\\[-2em]}%
    \inputminted[
        frame=lines,
        framesep=2mm,
        bgcolor=LightGray!20!White,
        fontsize=\footnotesize,
        linenos,
        firstline=#4,
        lastline=#5,
      ]{#1}{#3}
  }
}
